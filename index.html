function doPost(e) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const data = JSON.parse(e.postData.contents);
  function getOrCreate(name) {
    return ss.getSheetByName(name) || ss.insertSheet(name);
  }
  const raw = getOrCreate("raw");
  const biz = getOrCreate("経費");
  if (raw.getLastRow() === 0) {
    raw.appendRow(["取引日","登録日時","タイプ","金額","個人負担額",
      "メモ","計上先","領収書","個人カテゴリ","事業カテゴリ","収入カテゴリ"]);
  }
  if (biz.getLastRow() === 0) {
    biz.appendRow(["取引日","登録日時","金額","メモ","領収書","経費カテゴリ"]);
  }
  raw.appendRow([data.txDate,data.date,data.type,data.amount,
    data.personalAmount||"",data.memo,data.alloc||"",data.receipt||"",
    (data.personalCats||[]).join(","),
    (data.bizCats||[]).join(","),
    (data.incomeCats||[]).join(",")]);
  if(data.bizCats&&data.bizCats.length>0){
    biz.appendRow([data.txDate,data.date,data.amount,
      data.memo,data.receipt||"",data.bizCats.join(",")]);
  }
  return ContentService.createTextOutput(JSON.stringify({ok:true}))
    .setMimeType(ContentService.MimeType.JSON);
}
function doGet(){
  return ContentService.createTextOutput(JSON.stringify({ok:true}))
    .setMimeType(ContentService.MimeType.JSON);
}
